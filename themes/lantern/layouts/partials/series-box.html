Usage in front matter: series: ["Building a Platform"] seriesOrder: 2 */}} {{ $terms := .GetTerms "series" }} {{ with $terms }} {{ $term := index . 0 }} {{ $parts := sort $term.Pages "Params.seriesOrder" "asc" }} {{ $total := len $parts }} {{ $current := 0 }} {{ range $i, $p := $parts }}{{ if eq $p $ }}{{ $current = add $i 1 }}{{ end }}{{ end }}

{{ $term.Page.Title | default $term.LinkTitle | default $term.Title }}

    {{ range $i, $p := $parts }} {{ $num := printf "%02d" (add $i 1) }} {{ if eq $p $ }}
  1. {{ $num }}{{ .Title }}← you are here
    2. {{ else }}
  2. {{ $num }}{{ .Title }}
    3. {{ end }} {{ end }}
{{ end }}
